## 单链表 (常用于邻接表->存储树和图)

数组模拟单链表,e[N]表示某个点的值,ne[N]表示某个点的next指针

![image-20220303234901229](https://test-1306598008.cos.ap-beijing.myqcloud.com/image-20220303234901229.png)

```cpp
#include <iostream>

using namespace std;

const int N = 1e5 + 10;

//head头指针的下标
//e[N]节点的值
//ne[N]节点的next
//idx当前用到了那个点
int head, e[N], ne[N], idx;

void init(){
    head = -1;
    idx = 0;
}

void adh(int x){
    e[idx] = x;
    ne[idx] = head;
    head = idx;
    idx++;
}

void del(int k){
    ne[k] = ne[ne[k]];
}

void adk(int k, int x){
    e[idx] = x;
    ne[idx] = ne[k];
    ne[k] = idx;
    idx++;
}

int main(){
    ios::sync_with_stdio(false);
    int M, k, x;
    cin >> M;

    init();

    for(int i = 0; i < M; i ++ ){
        char c;
        cin >> c;
        switch(c){
            case 'H':
                cin >> x;
                adh(x); break;
            case 'D':
                cin >> k;
                if (k == 0) head = ne[head];
                else del(k - 1);
                break;
            case 'I':
                cin >> k >> x;
                adk(k - 1, x); break;
        }

    }

    for(int i = head; i != -1; i = ne[i]) cout << e[i] << ' ' ;
    cout << endl;

    return 0;
}
```

## 双链表 (常用于优化某些问题)

```cpp
#include <iostream>
#include <cstring>

using namespace std;

const int N = 1e5 + 10;

int e[N], l[N], r[N], idx;

void init(){
    r[0] = 1, l[0] = 0;
    idx = 2;
}

void del(int k){
    r[l[k]] = r[k];
    l[r[k]] = l[k];
}

void add(int k, int x){
    e[idx] = x;
    r[idx] = r[k];
    l[idx] = k;
    l[r[k]] = idx;
    r[k] = idx;
    idx++;
}

int main(){
    ios::sync_with_stdio(false);
    int M, k, x;
    cin >> M;
    
    init();
    
    while (M -- ){
        string s;
        cin >> s;
        if(s == "L"){
            cin >> x;
            add(0, x);
        }
        else if(s == "R"){
            cin >> x;
            add(l[1], x);
        }
        else if(s == "D"){
            cin >> k;
            del(k + 1);
        }
        else if(s == "IL"){
            cin >> k >> x;
            add(l[k + 1], x);
        }
        else if(s == "IR"){
            cin >> k >> x;
            add(k + 1, x);
        }
    }
    
    for (int i = r[0]; i != 1; i = r[i] ) cout << e[i] << ' ';
    cout << endl;
    
    return 0;
}

```
